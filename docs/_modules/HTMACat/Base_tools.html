### Function1: a is inclued in b return True

[docs]def isinclude(a, b):
    item = []
    for i in a:
        if i not in b:
            item += [False]
            break
        elif i in b:
            item += [True]
    if len(a) == item.count(True):
        return True
    else:
        return False

[docs]def isequal(a, b):
    item = []
    if len(a) == len(b):
        for i in range(len(a)):
            if a[i] == b[i]:
                item += [True]
            else:
                item += [False]
        if len(a) == item.count(True):
            return True
        else:
            return False
    else:
        return False
